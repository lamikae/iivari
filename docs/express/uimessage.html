UIMessage is a small piece of information on the screen.
The slide title, notice about background action etc.
             		                            
class UIMessage

    constructor: (options) ->
        @container = $(".#{options.container_class}")
        unless @container
            console.log "No container_class"
            return false

        @container.css
            width: "auto"
            height: "auto"

        if options.css
            @css options.css

        @visible = options.visible
        @visible ?= true
        unless @visible
            @container.hide()

        if options.html
            @container.html(options.html)

    css: (params) =>
        @container.css params
        return this

    show: (text, delay) =>
        @container.text(text)
        return unless @visible

        @visible = false
    unhide: =>
        @container.fadeIn("fast")

$.fn.uimessage = (options) ->
    options = $.fn.uimessage.options = $.extend $.fn.uimessage.options, options
    return new UIMessage(options)
