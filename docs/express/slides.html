    find_images = (path) ->
        console.log "Index images from #{path}"
        t0 = Date.now()

        glob_options = {}
        images = _.select glob.sync("#{path}/**/*", glob_options), (file) ->
            file.match(/jpe?g/i)

        t1 = Date.now()
        dt = (t1-t0)/1000
        console.log "Indexed #{_.size images} images in #{Math.floor(dt / 60)} min #{(dt % 60).toPrecision(2)} sec"
        return images
